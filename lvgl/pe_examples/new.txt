// 生成随机数字块（示例：生成“2”）
void spawn_random_tile(void)
{
	// 逻辑：在空位置随机选择一个格子，设置为数字2
	int x, y;
	// 这里需要实现“找空位置”的逻辑，示例中简化为固定位置（0,0）
	x = 0;
	y = 0;
	if (matrix[x][y] == 0)
	{
		matrix[x][y] = 2;
		// 移除空样式，添加数字2的样式
		lv_obj_remove_style(tiles[x][y], &style_tile_0, 0);
		lv_obj_add_style(tiles[x][y], &style_tile_2, 0);
		// 更新文字
		lv_obj_t *label = lv_label_create(tiles[x][y]);
		lv_label_set_text(label, "2");
		lv_obj_center(label);
	}
}
